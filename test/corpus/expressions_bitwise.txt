==================
Bitwise operator precedence
==================
// Should parse as: (a & b) ^ (c | d)
// This is incorrect based on C-like precedence.
// Correct is: a | (b ^ (c & d)). Precedence is &, ^, |
contract C {
  function f() {
    uint a = 0x1 | 0x2 ^ 0x4 & 0x8;
  }
}

    uint a = 1 << 2 & 3;
